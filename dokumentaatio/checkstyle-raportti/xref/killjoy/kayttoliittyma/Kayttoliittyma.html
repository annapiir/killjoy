40      private void luoKomponentit(Container container) {

42          container.setLayout(layout);
43  
44          //Luodaan käyttöliittymään tietojen syöttämistä ja esittämistä varten paneelit
45          JPanel pvmPaneeli = luoPvmPaneeli();

57          tietoPaneeli.add(tietoAlkoholi);
58          tietoPaneeli.add(tietoMuuKulu);

60          luoRajat(tietoPaneeli, "Juomasi alkoholi");

62          //ALKOHOLIPANEELI
63          JPanel alkoholiPaneeli = new JPanel();

65          alkoholiPaneeli.setLayout(alkoholiLayout);
66  
67          //Luodaan valikot

70          JComboBox<Integer> maaraValikko = new JComboBox<>(maaraLista());

72          JComboBox<Double> kokoValikko = new JComboBox<>(alkoholiKokoLista());

75          JComboBox<Double> vahvuusValikko = new JComboBox<>(alkoholiVahvuusLista());

79          hintaValikko.setEditable(true);
80          JLabel hintaTeksti2 = new JLabel("euroa ");
81          //ja nappi tulosten tallettamiselle
82          JButton lisaaAlkoholi = new JButton("Lisää");
83          AlkoholiKuuntelija alkoholiKuuntelija = new AlkoholiKuuntelija(this.kayttokerta, maaraValikko, kokoValikko, vahvuusValikko, hintaValikko, lisaaAlkoholi, tietoAlkoholi, maaraLista(), hintaLista(), alkoholiKokoLista(), alkoholiVahvuusLista(), tietoMuuKulu);
84          lisaaAlkoholi.addActionListener(alkoholiKuuntelija);
85  
86          luoRajat(alkoholiPaneeli, "Lisää alkoholiannoksia");
87  
88          alkoholiPaneeli.add(maaraValikko);
89          alkoholiPaneeli.add(maaraTeksti);
90          alkoholiPaneeli.add(kokoValikko);
91          alkoholiPaneeli.add(vahvuusTeksti);
92          alkoholiPaneeli.add(vahvuusValikko);
93          alkoholiPaneeli.add(hintaTeksti);
94          alkoholiPaneeli.add(hintaValikko);
95          alkoholiPaneeli.add(hintaTeksti2);
96          alkoholiPaneeli.add(lisaaAlkoholi);
97  
98          //KULUPANEELI

100         BoxLayout kuluLayout = new BoxLayout(kuluPaneeli, BoxLayout.X_AXIS);
101         kuluPaneeli.setLayout(kuluLayout);
102 
103         //Luodaan valikot
104         //Määrälle
105         JLabel maaraKuluTeksti = new JLabel(" x ");
106         JComboBox<Integer> maaraKuluValikko = new JComboBox<>(maaraLista());
107         //Hinnalle
108         JLabel hintaKuluTeksti = new JLabel(" hinta ");
109         JComboBox<Double> hintaKuluValikko = new JComboBox<>(hintaLista());
110         hintaKuluValikko.setEditable(true);
111         JLabel hintaKuluTeksti2 = new JLabel("euroa  ");
112         //ja nappi tulosten tallettamiselle
113         JButton lisaaKulu = new JButton("Lisää");
114         KuluKuuntelija kuluKuuntelija;
115         kuluKuuntelija = new KuluKuuntelija(this.kayttokerta, maaraKuluValikko, hintaKuluValikko, lisaaKulu, maaraLista(), hintaLista(), tietoMuuKulu);
116         lisaaKulu.addActionListener(kuluKuuntelija);
117 
118         luoRajat(kuluPaneeli, "Lisää muita kuluja");
119 
120         kuluPaneeli.add(maaraKuluValikko);
121         kuluPaneeli.add(maaraKuluTeksti);
122         kuluPaneeli.add(hintaKuluTeksti);
123         kuluPaneeli.add(hintaKuluValikko);
124         kuluPaneeli.add(hintaKuluTeksti2);
125         kuluPaneeli.add(lisaaKulu);
126 
127         //Lisätään kaikki osaset ikkunaan
128         container.add(pvmPaneeli);
129         container.add(alkoholiPaneeli);
130         container.add(kuluPaneeli);
131         container.add(tietoPaneeli);
132         container.add(laskeTulokset);
133         container.add(tuloksetTeksti);
