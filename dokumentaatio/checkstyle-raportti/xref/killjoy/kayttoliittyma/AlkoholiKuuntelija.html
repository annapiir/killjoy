6   package killjoy.kayttoliittyma;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import javax.swing.JButton;
11  import javax.swing.JComboBox;
12  import javax.swing.JTextArea;
13  import killjoy.logiikka.KayttokertaIF;
14  
15  /**
16   *
17   * @author annapiir
18   */
19  public class AlkoholiKuuntelija implements ActionListener {
20  
21      private KayttokertaIF kayttokerta;
22      private JComboBox<Integer> maaraValikko;
23      private JComboBox<Double> kokoValikko;
24      private JComboBox<Double> vahvuusValikko;
25      private JComboBox<Double> hintaValikko;
26      private JButton lisaaAlkoholi;
27      private JTextArea tietoAlkoholi;
28      private Integer[] maaraLista;
29      private Double[] hintaLista;
30      private Double[] alkoholiKokoLista;
31      private Double[] alkoholiVahvuusLista;
32      private JTextArea tietoMuuKulu;
33  
34      AlkoholiKuuntelija(KayttokertaIF kayttokerta, JComboBox<Integer> maaraValikko, JComboBox<Double> kokoValikko, JComboBox<Double> vahvuusValikko, JComboBox<Double> hintaValikko, JButton lisaaAlkoholi, JTextArea tietoAlkoholi, Integer[] maaraLista, Double[] hintaLista, Double[] alkoholiKokoLista, Double[] alkoholiVahvuusLista, JTextArea tietoMuuKulu) {
35          this.kayttokerta = kayttokerta;
36          this.maaraValikko = maaraValikko;
37          this.kokoValikko = kokoValikko;
38          this.vahvuusValikko = vahvuusValikko;
39          this.hintaValikko = hintaValikko;
40          this.lisaaAlkoholi = lisaaAlkoholi;
41          this.tietoAlkoholi = tietoAlkoholi;
42          this.maaraLista = maaraLista;
43          this.hintaLista = hintaLista;
44          this.alkoholiKokoLista = alkoholiKokoLista;
45          this.alkoholiVahvuusLista = alkoholiVahvuusLista;
46          this.tietoMuuKulu = tietoMuuKulu;
47      }
48  
49      @Override
50      public void actionPerformed(ActionEvent e) {
51          int maara = this.maaraLista[this.maaraValikko.getSelectedIndex()];
52          double koko = this.alkoholiKokoLista[this.kokoValikko.getSelectedIndex()];
53          double vahvuus = this.alkoholiVahvuusLista[this.vahvuusValikko.getSelectedIndex()];
54          double hinta = this.hintaLista[this.hintaValikko.getSelectedIndex()];
55          this.kayttokerta.lisaaKertaAnnos(hinta, maara, vahvuus, koko);
56  
57          this.tietoAlkoholi.setText(this.kayttokerta.tulostaAlkoholit());
58          this.tietoMuuKulu.setText(this.kayttokerta.tulostaKulut());
59  
60      }
61  
62  }
