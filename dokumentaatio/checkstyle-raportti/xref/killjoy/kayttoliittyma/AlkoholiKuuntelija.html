1   package killjoy.kayttoliittyma;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.JButton;
6   import javax.swing.JComboBox;
7   import javax.swing.JTextArea;
8   import killjoy.logiikka.KayttokertaIF;
9   
10  /**
11   * Luokalla toteutetaan alhkoliannosten tallentamiseen liittyviä toimia.
12   *
13   * @see killjoy.kayttoliittyma.Kayttoliittyma
14   * @see killjoy.logiikka.Kayttokerta
15   */
16  public class AlkoholiKuuntelija implements ActionListener {
17  
18      private KayttokertaIF kayttokerta;
19      private JComboBox<Integer> maaraValikko;
20      private JComboBox<Double> kokoValikko;
21      private JComboBox<Double> vahvuusValikko;
22      private JComboBox<Double> hintaValikko;
23      private JButton lisaaAlkoholi;
24      private JTextArea tietoAlkoholi;
25      private Integer[] maaraLista;
26      private Double[] hintaLista;
27      private Double[] alkoholiKokoLista;
28      private Double[] alkoholiVahvuusLista;
29      private JTextArea tietoMuuKulu;
30  
31      /**
32       * Konstruktori saa parametreiksi alkoholiannoksen tallentamiseen liittyvät
33       * valikot ja listat, napin, viitteen ohjelmalogiikkaa hallinnoivaan
34       * käyttökertaan sekä tulosten esittämistä varten tuloslaatikon.
35       *
36       * @param kayttokerta
37       * @param maaraValikko
38       * @param kokoValikko
39       * @param vahvuusValikko
40       * @param hintaValikko
41       * @param lisaaAlkoholi
42       * @param tietoAlkoholi
43       * @param maaraLista
44       * @param hintaLista
45       * @param alkoholiKokoLista
46       * @param alkoholiVahvuusLista
47       * @param tietoMuuKulu
48       */
49      AlkoholiKuuntelija(KayttokertaIF kayttokerta, JComboBox<Integer> maaraValikko, JComboBox<Double> kokoValikko, JComboBox<Double> vahvuusValikko, JComboBox<Double> hintaValikko, JButton lisaaAlkoholi, JTextArea tietoAlkoholi, Integer[] maaraLista, Double[] hintaLista, Double[] alkoholiKokoLista, Double[] alkoholiVahvuusLista, JTextArea tietoMuuKulu) {
50          this.kayttokerta = kayttokerta;
51          this.maaraValikko = maaraValikko;
52          this.kokoValikko = kokoValikko;
53          this.vahvuusValikko = vahvuusValikko;
54          this.hintaValikko = hintaValikko;
55          this.lisaaAlkoholi = lisaaAlkoholi;
56          this.tietoAlkoholi = tietoAlkoholi;
57          this.maaraLista = maaraLista;
58          this.hintaLista = hintaLista;
59          this.alkoholiKokoLista = alkoholiKokoLista;
60          this.alkoholiVahvuusLista = alkoholiVahvuusLista;
61          this.tietoMuuKulu = tietoMuuKulu;
62      }
63  
64      @Override
65      public void actionPerformed(ActionEvent e) {
66          try {
67              int maara = this.maaraLista[this.maaraValikko.getSelectedIndex()];
68              double koko = this.alkoholiKokoLista[this.kokoValikko.getSelectedIndex()];
69              double vahvuus = this.alkoholiVahvuusLista[this.vahvuusValikko.getSelectedIndex()];
70              double hinta = this.hintaLista[this.hintaValikko.getSelectedIndex()];
71              this.kayttokerta.lisaaKertaAnnos(hinta, maara, vahvuus, koko);
72  
73              this.tietoAlkoholi.setText(this.kayttokerta.tulostaAlkoholit());
74              this.tietoMuuKulu.setText(this.kayttokerta.tulostaKulut());
75          } catch (Exception exception) {
76  
77          }
78  
79      }
80  
81  }
