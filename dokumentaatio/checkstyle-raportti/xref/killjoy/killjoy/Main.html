6   package killjoy.killjoy;
7   
8   import killjoy.kayttoliittyma.Tiedonlukija;
9   import killjoy.logiikka.Kayttokerta;
10  import killjoy.logiikka.KertaAnnos;
11  
12  /**
13   *
14   * @author annapiir
15   */
16  public class Main {
17  
18      public static void main(String[] args) {
19          //Tulostetaan tervehdys
20          System.out.println("Tervetuloa KillJoy-alkoholilaskuriin!");
21  
22          //Luodaan lukija tiedoille ja uusi käyttökerta
23          Tiedonlukija tiedonlukija = new Tiedonlukija();
24          Kayttokerta kerta = tiedonlukija.luoKayttokerta();
25  
26          //Luodaan lista käyttökerran juomisista
27          System.out.println("Mitä joit?");
28  
29          //Lisätään juomiset ArrayListiin (lopetusehtoa pitää vielä katsoa)
30          while (true) {
31              kerta.lisaaKertaAnnos(tiedonlukija.luoKertaAnnos(kerta));
32  
33              if (tiedonlukija.lopeta()) {
34                  break;
35              }
36          }
37          
38          System.out.println("\nSyötä seuraavaksi muut illanvietton kulut:");
39          
40     
41          if(tiedonlukija.onkoMuitaKuluja()) {
42              //Lisätään muut kulut ArrayListiin
43              while (true) {
44                  kerta.lisaaMuuKulu(tiedonlukija.luoMuuKulu(kerta));
45  
46                  if (tiedonlukija.lopeta()) {
47                      break;
48                  }
49              }
50          }
51  
52  
53          //Testataan lopputulosta tulostamalla käyttökerran tiedot 
54          System.out.println(kerta);
55  
56      }
57  }
