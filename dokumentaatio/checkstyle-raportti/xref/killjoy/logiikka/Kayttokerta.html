1   package killjoy.logiikka;
2   
3   import java.util.ArrayList;
4   
5   /**
6    * Luokka pitää yllä listaa käyttökertaan liittyvistä alkoholeista ja muista
7    * kuluista, ja tekee käyttökertaan liittyvät laskelmat. Luokan kautta
8    * kommunikoidaan käyttöliittmän kanssa.
9    */
10  public class Kayttokerta implements KayttokertaIF {
11  
12      private String pvm;
13      private ArrayList<KertaAnnos> alkoholit;
14      private ArrayList<Kulu> kulut;
15  
16      /**
17       * Konstruktori luo uuden käyttökerran. Käyttökertaan liittyvät päivämäärä
18       * sekä listat alkoholien ja kulujen säilytystä varten.
19       *
20       */
21      public Kayttokerta() {
22          this.pvm = null;
23          this.alkoholit = new ArrayList<>();
24          this.kulut = new ArrayList<>();
25      }
26  
27      /**
28       * Palauttaa päivämäärän.
29       *
30       * @return Päivämäärä (String)
31       */
32      public String getPvm() {
33          return pvm;
34      }
35  
36      /**
37       * Asettaa päivämäärän.
38       *
39       * @param pvm (String)
40       */
41      public void setPvm(String pvm) {
42          this.pvm = pvm;
43      }
44  
45      /**
46       * Palauttaa listan käyttökertaan liittyvistä alkoholiannoksista.
47       *
48       * @return Lista alkoholiannoksista (ArrayList)
49       */
50      public ArrayList<KertaAnnos> getAlkoholit() {
51          return alkoholit;
52      }
53  
54      /**
55       * Metodi lisää uuden alkoholiannoksen alkoholilistaan ja annoksesta
56       * syntyvät kulut kuluihin. Parametrina annetaan yksi kerta-annos.
57       *
58       * @param hinta
59       * @param maara
60       * @param vahvuus
61       * @param annoskoko
62       */
63      public void lisaaKertaAnnos(double hinta, int maara, double vahvuus, double annoskoko) {
64          this.alkoholit.add(new KertaAnnos(hinta, maara, vahvuus, annoskoko));
65          this.kulut.add(new Kulu(hinta, maara) {
66          });
67      }
68  
69      /**
70       * Metodi lisää muun kuin alkoholista koituvan kulun kululistaan.
71       * Parametrina annetaan kulu.
72       *
73       * @param hinta
74       * @param maara
75       */
76      public void lisaaMuuKulu(double hinta, int maara) {
77          this.kulut.add(new Kulu(hinta, maara) {
78          });
79      }
80  
81      /**
82       * Metodi laskee käyttökertaan liittyvän alkoholimäärän. Lasketaan
83       * summaamalla yksittäisiin kerta-annoksiin liittyvät alkoholimäärät.
84       *
85       * @return alkoholimäärä
86       */
87      public double laskeAlkoholiKayttokerrasta() {
88          double maaraYht = 0;
89  
90          for (KertaAnnos kertaAnnos : this.alkoholit) {
91              maaraYht = +kertaAnnos.laskeAlkoholi();
92          }
93  
94          return maaraYht;
95      }
96  
97      /**
98       * Metodi laskee kulut yhdestä käyttökerrasta summaamalla kululistan kulut.
99       *
100      * @return kulut yhteensä
101      */
102     public double laskeKulut() {
103         double kulut = 0;
104 
105         for (Kulu kulu : this.kulut) {
106             kulut += kulu.hintaYht();
107         }
108 
109         return kulut;
110     }
111 
112     /**
113      * Metodi laskee, monta litraa alkoholipitoista nestettä tuli nautittua
114      * käyttökerran aikana.
115      *
116      * @return käyttökertaan liittyvä alkoholinkulutus litroina
117      */
118     public double laskeAlkoholinKokonaiskulutus() {
119         double litroja = 0;
120 
121         for (KertaAnnos kertaAnnos : this.alkoholit) {
122             litroja = +kertaAnnos.laskeLitrat();
123         }
124 
125         return litroja;
126     }
127 
128     /**
129      * Tulostaa käyttökertaan liittyvien tietojen listan (vain
130      * tekstikäyttöliittymää varten).
131      *
132      * @return Käyttökertaan liittyvät alkoholit ja kulut (String)
133      */
134     @Override
135     public String toString() {
136         String saldo = "";
137         if (this.alkoholit.isEmpty() && this.kulut.isEmpty()) {
138             saldo = "ei alkoholia eikä muita kuluja";
139         } else if (this.alkoholit.isEmpty()) {
140             System.out.println("Et nauttinut lainkaan alkoholia!");
141         } else {
142             saldo += this.tulostaAlkoholit();
143         }
144 
145         return "Illan " + this.pvm + " saldo:\n" + saldo + "\nPuhdasta alkoholia joit " + this.laskeAlkoholiKayttokerrasta() + "l\nKaikkiaan kuluja koitui " + this.laskeKulut() + " euroa";
146     }
147 
148     /**
149      * Palauttaa käyttökertaan liittyvien alkoholien listan tulosteena.
150      *
151      * @return Alkoholit (String)
152      */
153     public String tulostaAlkoholit() {
154         String alkoholit = "Nautittu alkoholi:\n";
155         for (KertaAnnos annos : this.alkoholit) {
156             alkoholit = alkoholit + " - " + annos.toString() + "\n";
157         }
158 
159         return alkoholit + "\n";
160     }
161 
162     /**
163      * Palauttaa käyttökertaan liittyvien kulujen listan (alkoholi + muut kulut)
164      * tulosteena.
165      *
166      * @return Lista kuluista (String)
167      */
168     public String tulostaKulut() {
169         String kulut = "Kulut:\n";
170         for (Kulu kulu : this.kulut) {
171             kulut = kulut + " - " + kulu.toString() + "\n";
172         }
173 
174         return kulut;
175     }
176 
177 }
